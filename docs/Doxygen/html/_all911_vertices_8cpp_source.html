50void All911Vertices::createAllVertices(Layout *layout)
51{
52 vector<int> psapList;
53 vector<int> respList;
54 psapList.clear();
55 respList.clear();
56
57 int callersPerZone[] = {0, 0, 0, 0};
58 int respPerZone[] = {0, 0, 0, 0};
59
60 Layout911 *layout911 = dynamic_cast<Layout911 *>(layout);
61
62 for (int i = 0; i < Simulator::getInstance().getTotalVertices(); i++) {
63 // Create all callers
64 if (layout->vertexTypeMap_[i] == CALR) {
65 callNum_[i] = initRNG.inRange(callNumRange_[0], callNumRange_[1]);
66 callersPerZone[layout911->zone(i)] += callNum_[i];
67 }
68
69 // Find all PSAPs
70 if (layout->vertexTypeMap_[i] == PSAP) {
71 psapList.push_back(i);
72 }
73
74 // Find all resps
75 if (layout->vertexTypeMap_[i] == RESP) {
76 respList.push_back(i);
77 respPerZone[layout911->zone(i)] += 1;
78 }
79 }
80
81 // Create all psaps
82 // Dispatchers in a psap = [callers in the zone * k] + some randomness
83 for (int i = 0; i < psapList.size(); i++) {
84 int psapQ = layout911->zone(i);
85 int dispCount = (callersPerZone[psapQ] * dispNumScale_) + initRNG.inRange(-5, 5);
86 if (dispCount < 1) {
87 dispCount = 1;
88 }
89 dispNum_[psapList[i]] = dispCount;
90 }
91
92 // Create all responders
93 // Responders in a node = [callers in the zone * k]/[number of responder nodes] + some randomness
94 for (int i = 0; i < respList.size(); i++) {
95 int respQ = layout911->zone(respList[i]);
96 int respCount
97 = (callersPerZone[respQ] * respNumScale_) / respPerZone[respQ] + initRNG.inRange(-5, 5);
98 if (respCount < 1) {
99 respCount = 1;
100 }
101 respNum_[respList[i]] = respCount;
102 }
103}
