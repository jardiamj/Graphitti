148void Hdf5GrowthRecorder::writeRadiiRates()
149{
150 try {
151 // Write radii and rates histories information:
152 hsize_t offset[2], count[2];
153 hsize_t dimsm[2];
154 DataSpace *dataspace;
155 DataSpace *memspace;
156
157 // write radii history
158 offset[0] = Simulator::getInstance().getCurrentStep();
159 offset[1] = 0;
160 count[0] = 1;
161 count[1] = Simulator::getInstance().getTotalVertices();
162 dimsm[0] = 1;
163 dimsm[1] = Simulator::getInstance().getTotalVertices();
164 memspace = new DataSpace(2, dimsm, nullptr);
165 dataspace = new DataSpace(dataSetRadiiHist_->getSpace());
166 dataspace->selectHyperslab(H5S_SELECT_SET, count, offset);
167 dataSetRadiiHist_->write(radiiHistory_, H5_FLOAT, *memspace, *dataspace);
168 delete dataspace;
169 delete memspace;
170
171 // write rates history
172 offset[0] = Simulator::getInstance().getCurrentStep();
173 offset[1] = 0;
174 count[0] = 1;
175 count[1] = Simulator::getInstance().getTotalVertices();
176 dimsm[0] = 1;
177 dimsm[1] = Simulator::getInstance().getTotalVertices();
178 memspace = new DataSpace(2, dimsm, nullptr);
179 dataspace = new DataSpace(dataSetRadiiHist_->getSpace());
180 dataspace->selectHyperslab(H5S_SELECT_SET, count, offset);
181 dataSetRatesHist_->write(ratesHistory_, H5_FLOAT, *memspace, *dataspace);
182 delete dataspace;
183 delete memspace;
184 }
185
186 // catch failure caused by the H5File operations
