62void XmlSTDPRecorder::getValues()
63{
64 Simulator &simulator = Simulator::getInstance();
65 shared_ptr<Connections> connections = simulator.getModel()->getConnections();
66 AllEdges &synapses = (*connections->getEdges());
67 const int currentStep = simulator.getCurrentStep();
68
69 for (int i = 0; i < simulator.getTotalVertices(); i++) {
70 synapses.W_[i] = weightsHistory_[currentStep][i];
71 synapses.sourceVertexIndex_[i] = sourceNeuronIndexHistory_[currentStep][i];
72 synapses.destVertexIndex_[i] = destNeuronIndexHistory_[currentStep][i];

145void XmlSTDPRecorder::saveSimData(const AllVertices &neurons)
146{
147 Simulator &simulator = Simulator::getInstance();
148
149 // create Neuron Types matrix
150 VectorMatrix neuronTypes(MATRIX_TYPE, MATRIX_INIT, 1, simulator.getTotalVertices(), EXC);
151 for (int i = 0; i < simulator.getTotalVertices(); i++) {
152 neuronTypes[i] = simulator.getModel()->getLayout()->vertexTypeMap_[i];
153 }
154
155 // create neuron threshold matrix
156 VectorMatrix neuronThresh(MATRIX_TYPE, MATRIX_INIT, 1, simulator.getTotalVertices(), 0);
157 for (int i = 0; i < simulator.getTotalVertices(); i++) {
158 neuronThresh[i] = dynamic_cast<const AllIFNeurons &>(neurons).Vthresh_[i];
159 }
160
161 // Write XML header information:
162 resultOut_ << "<?xml version=\"1.0\" standalone=\"no\"?>\n"
163 << "<!-- State output file for the DCT growth modeling-->\n";
164 //stateOut << version; TODO: version
165
166 // Write the core state information:
167 resultOut_ << "<SimState>\n";
168 resultOut_ << " " << toXML("sourceNeuronIndexHistory", sourceNeuronIndexHistory_) << endl;
169 resultOut_ << " " << toXML("destNeuronIndexHistory", destNeuronIndexHistory_) << endl;
170 resultOut_ << " " << toXML("weightsHistory", weightsHistory_) << endl;
171 resultOut_ << " " << burstinessHist_.toXML("burstinessHist") << endl;
172 resultOut_ << " " << spikesHistory_.toXML("spikesHistory") << endl;
173 resultOut_ << " " << simulator.getModel()->getLayout()->xloc_->toXML("xloc") << endl;
174 resultOut_ << " " << simulator.getModel()->getLayout()->yloc_->toXML("yloc") << endl;
175 resultOut_ << " " << neuronTypes.toXML("neuronTypes") << endl;
176
177 // create starter neuron matrix
178 int num_starter_neurons
179 = static_cast<int>(simulator.getModel()->getLayout()->numEndogenouslyActiveNeurons_);
180 if (num_starter_neurons > 0) {
181 VectorMatrix starterNeurons(MATRIX_TYPE, MATRIX_INIT, 1, num_starter_neurons);
182 getStarterNeuronMatrix(starterNeurons, simulator.getModel()->getLayout()->starterMap_);
183 resultOut_ << " " << starterNeurons.toXML("starterNeurons") << endl;
184 }
185
186 // Write neuron thresold
187 resultOut_ << " " << neuronThresh.toXML("neuronThresh") << endl;
188
189 // write time between growth cycles
190 resultOut_
191 << " <Matrix name=\"Tsim\" type=\"complete\" rows=\"1\" columns=\"1\" multiplier=\"1.0\">"
192 << endl;
193 resultOut_ << " " << simulator.getEpochDuration() << endl;
194 resultOut_ << "</Matrix>" << endl;
195
196 // write simulation end time
197 resultOut_
198 << " <Matrix name=\"simulationEndTime\" type=\"complete\" rows=\"1\" columns=\"1\" multiplier=\"1.0\">"
199 << endl;
200 resultOut_ << " " << g_simulationStep * simulator.getDeltaT() << endl;
201 resultOut_ << "</Matrix>" << endl;
202 resultOut_ << "</SimState>" << endl;
203}
