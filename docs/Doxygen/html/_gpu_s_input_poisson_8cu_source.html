164__global__ void inputStimulusDevice( int n, int* nISIs_d, bool* masks_d, BGFLOAT deltaT, BGFLOAT lambda, curandState* devStates_d, AllDSSynapsesDeviceProperties* allEdgesDevice )
165{
166 int idx = blockIdx.x * blockDim.x + threadIdx.x;
167 if ( idx >= n )
168 return;
169
170 if (masks_d[idx] == false)
171 return;
172
173 BGSIZE iEdg = idx;
174
175 int rnISIs = nISIs_d[idx]; // load the value to a register
176 if (--rnISIs <= 0)
177 {
178 // add a spike
179 uint32_t &delay_queue = allEdgesDevice->delayQueue[iEdg];
180 int delayIdx = allEdgesDevice->delayIdx[iEdg];
181 int ldelayQueue = allEdgesDevice->ldelayQueue[iEdg];
182 int total_delay = allEdgesDevice->total_delay[iEdg];
183
184 // Add to spike queue
185
186 // calculate index where to insert the spike into delayQueue
187 int idx = delayIdx + total_delay;
188 if ( idx >= ldelayQueue ) {
189 idx -= ldelayQueue;
190 }
191
192 // set a spike
193 //assert( !(delay_queue[0] & (0x1 << idx)) );
194 delay_queue |= (0x1 << idx);
195
196 // update interval counter (exponectially distribution ISIs, Poisson)
197 curandState localState = devStates_d[idx];
198
199 BGFLOAT isi = -lambda * log(curand_uniform( &localState ));
200 // delete isi within refractoriness
201 while (curand_uniform( &localState ) <= exp(-(isi*isi)/32))
202 isi = -lambda * log(curand_uniform( &localState ));
203 // convert isi from msec to steps
204 rnISIs = static_cast<int>( (isi / 1000) / deltaT + 0.5 );
205 devStates_d[idx] = localState;
206 }
207 nISIs_d[idx] = rnISIs;
208}
