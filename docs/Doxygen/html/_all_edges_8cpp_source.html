103void AllEdges::setupEdges(const int numVertices, const int maxEdges)
104{
105 BGSIZE maxTotalEdges = maxEdges * numVertices;
106
107 maxEdgesPerVertex_ = maxEdges;
108 totalEdgeCount_ = 0;
109 countVertices_ = numVertices;
110
111 if (maxTotalEdges != 0) {
112 destVertexIndex_ = new int[maxTotalEdges];
113 W_ = new BGFLOAT[maxTotalEdges];
114 summationPoint_ = new BGFLOAT *[maxTotalEdges];
115 sourceVertexIndex_ = new int[maxTotalEdges];
116 type_ = new edgeType[maxTotalEdges];
117 inUse_ = new bool[maxTotalEdges];
118 edgeCounts_ = new BGSIZE[numVertices];
119
120 for (BGSIZE i = 0; i < maxTotalEdges; i++) {
121 summationPoint_[i] = nullptr;
122 inUse_[i] = false;
123 W_[i] = 0;
124 }
125
126 for (int i = 0; i < numVertices; i++) {
127 edgeCounts_[i] = 0;
128 }
129 }

197void AllEdges::createEdgeIndexMap(shared_ptr<EdgeIndexMap> edgeIndexMap)
198{
199 Simulator &simulator = Simulator::getInstance();
200 int vertexCount = simulator.getTotalVertices();
201 int totalEdgeCount = 0;
202
203 // count the total edges
204 for (int i = 0; i < vertexCount; i++) {
205 assert(static_cast<int>(edgeCounts_[i]) < simulator.getMaxEdgesPerVertex());
206 totalEdgeCount += edgeCounts_[i];
207 }
208
209 LOG4CPLUS_TRACE(fileLogger_, endl
210 << "totalEdgeCount: in edgeIndexMap " << totalEdgeCount << endl);
211
212 // Create vector for edge forwarding map
213 vector<BGSIZE> rgEdgeEdgeIndexMap[vertexCount];
214
215 BGSIZE edg_i = 0;
216 int curr = 0;
217
218 LOG4CPLUS_TRACE(edgeLogger_,
219 "\nSize of edge Index Map " << vertexCount << "," << totalEdgeCount << endl);
220
221 for (int i = 0; i < vertexCount; i++) {
222 BGSIZE edge_count = 0;
223 edgeIndexMap->incomingEdgeBegin_[i] = curr;
224 for (int j = 0; j < simulator.getMaxEdgesPerVertex(); j++, edg_i++) {
225 if (inUse_[edg_i]) {
226 int idx = sourceVertexIndex_[edg_i];
227 rgEdgeEdgeIndexMap[idx].push_back(edg_i);
228
229 edgeIndexMap->incomingEdgeIndexMap_[curr] = edg_i;
230 curr++;
231 edge_count++;
232 }
233 }
234
235 if (edge_count != edgeCounts_[i]) {
236 LOG4CPLUS_FATAL(edgeLogger_,
237 "\nedge_count does not match edgeCounts_" << edge_count << endl);
238 throw runtime_error("createEdgeIndexMap: edge_count does not match edgeCounts_.");
239 }
240
241 edgeIndexMap->incomingEdgeCount_[i] = edge_count;
242 }
243
244 if (totalEdgeCount != curr) {
245 LOG4CPLUS_FATAL(edgeLogger_,
246 "Curr does not match the totalEdgeCount. curr are " << curr << endl);
247 throw runtime_error("createEdgeIndexMap: Curr does not match the totalEdgeCount.");
248 }
249 totalEdgeCount_ = totalEdgeCount;
250 LOG4CPLUS_DEBUG(edgeLogger_, endl << "totalEdgeCount: " << totalEdgeCount_ << endl);
251
252 edg_i = 0;
253 for (int i = 0; i < vertexCount; i++) {
254 edgeIndexMap->outgoingEdgeBegin_[i] = edg_i;
255 edgeIndexMap->outgoingEdgeCount_[i] = rgEdgeEdgeIndexMap[i].size();
256
257 for (BGSIZE j = 0; j < rgEdgeEdgeIndexMap[i].size(); j++, edg_i++) {
258 edgeIndexMap->outgoingEdgeIndexMap_[edg_i] = rgEdgeEdgeIndexMap[i][j];
259 }
260 }
261}

301void AllEdges::addEdge(BGSIZE &iEdg, edgeType type, const int srcVertex, const int destVertex,
302 BGFLOAT *sumPoint, const BGFLOAT deltaT)
303{
304 if (edgeCounts_[destVertex] >= maxEdgesPerVertex_) {
305 LOG4CPLUS_FATAL(edgeLogger_, "Vertex : " << destVertex << " ran out of space for new edges.");
306 throw runtime_error("Vertex : " + destVertex + string(" ran out of space for new edges."));
307 }
308
309 // add it to the list: find first edge location for vertex destVertex
310 // that isn't in use.
311 for (BGSIZE i = 0; i < maxEdgesPerVertex_; i++) {
312 iEdg = maxEdgesPerVertex_ * destVertex + i;
313 if (!inUse_[iEdg]) {
314 break;
315 }
316 }
317
318 edgeCounts_[destVertex]++;
319
320 // create an edge
321 createEdge(iEdg, srcVertex, destVertex, sumPoint, deltaT, type);
322}
