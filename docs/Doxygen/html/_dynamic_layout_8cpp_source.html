157edgeType DynamicLayout::edgType(const int srcVertex, const int destVertex)
158{
159 if (vertexTypeMap_[srcVertex] == INH && vertexTypeMap_[destVertex] == INH)
160 return II;
161 else if (vertexTypeMap_[srcVertex] == INH && vertexTypeMap_[destVertex] == EXC)
162 return IE;
163 else if (vertexTypeMap_[srcVertex] == EXC && vertexTypeMap_[destVertex] == INH)
164 return EI;
165 else if (vertexTypeMap_[srcVertex] == EXC && vertexTypeMap_[destVertex] == EXC)
166 return EE;
