117edgeType FixedLayout::edgType(const int srcVertex, const int destVertex)
118{
119 if (vertexTypeMap_[srcVertex] == INH && vertexTypeMap_[destVertex] == INH)
120 return II;
121 else if (vertexTypeMap_[srcVertex] == INH && vertexTypeMap_[destVertex] == EXC)
122 return IE;
123 else if (vertexTypeMap_[srcVertex] == EXC && vertexTypeMap_[destVertex] == INH)
124 return EI;
125 else if (vertexTypeMap_[srcVertex] == EXC && vertexTypeMap_[destVertex] == EXC)
126 return EE;
127
128 return ETYPE_UNDEF;
