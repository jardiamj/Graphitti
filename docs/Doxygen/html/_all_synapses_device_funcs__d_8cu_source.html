73__device__ void changeDSSynapsePSRDevice(AllDSSynapsesDeviceProperties* allEdgesDevice, const BGSIZE iEdg, const uint64_t simulationStep, const BGFLOAT deltaT)
74{
75 //assert( iEdg < allEdgesDevice->maxEdgesPerVertex * allEdgesDevice->countVertices_ );
76
77 uint64_t &lastSpike = allEdgesDevice->lastSpike_[iEdg];
78 BGFLOAT &r = allEdgesDevice->r_[iEdg];
79 BGFLOAT &u = allEdgesDevice->u_[iEdg];
80 BGFLOAT D = allEdgesDevice->D_[iEdg];
81 BGFLOAT F = allEdgesDevice->F_[iEdg];
82 BGFLOAT U = allEdgesDevice->U_[iEdg];
83 BGFLOAT W = allEdgesDevice->W_[iEdg];
84 BGFLOAT &psr = allEdgesDevice->psr_[iEdg];
85 BGFLOAT decay = allEdgesDevice->decay_[iEdg];
86
87 // adjust synapse parameters
88 if (lastSpike != ULONG_MAX) {
89 BGFLOAT isi = (simulationStep - lastSpike) * deltaT ;
90 r = 1 + ( r * ( 1 - u ) - 1 ) * exp( -isi / D );
91 u = U + u * ( 1 - U ) * exp( -isi / F );
92 }
93 psr += ( ( W / decay ) * u * r );// calculate psr
94 lastSpike = simulationStep; // record the time of the spike
95}

129__device__ void stdpLearningDevice(AllSTDPSynapsesDeviceProperties* allEdgesDevice, const BGSIZE iEdg, double delta, double epost, double epre)
130{
131 BGFLOAT STDPgap = allEdgesDevice->STDPgap_[iEdg];
132 BGFLOAT muneg = allEdgesDevice->muneg_[iEdg];
133 BGFLOAT mupos = allEdgesDevice->mupos_[iEdg];
134 BGFLOAT tauneg = allEdgesDevice->tauneg_[iEdg];
135 BGFLOAT taupos = allEdgesDevice->taupos_[iEdg];
136 BGFLOAT Aneg = allEdgesDevice->Aneg_[iEdg];
137 BGFLOAT Apos = allEdgesDevice->Apos_[iEdg];
138 BGFLOAT Wex = allEdgesDevice->Wex_[iEdg];
139 BGFLOAT &W = allEdgesDevice->W_[iEdg];
140 edgeType type = allEdgesDevice->type_[iEdg];
141 BGFLOAT dw;
142
143 if (delta < -STDPgap) {
144 // Depression
145 dw = pow(fabs(W) / Wex, muneg) * Aneg * exp(delta / tauneg); // normalize
146 } else if (delta > STDPgap) {
147 // Potentiation
148 dw = pow(fabs(Wex - fabs(W)) / Wex, mupos) * Apos * exp(-delta / taupos); // normalize
149 } else {
150 return;
151 }
152
153 // dw is the percentage change in synaptic strength; add 1.0 to become the scaling ratio
154 dw = 1.0 + dw * epre * epost;
155
156 // if scaling ratio is less than zero, set it to zero so this synapse, its strength is always zero
157 if (dw < 0) {
158 dw = 0;
159 }
160
161 // current weight multiplies dw (scaling ratio) to generate new weight
162 W *= dw;
163
164 // if new weight is bigger than Wex_ (maximum allowed weight), then set it to Wex_
165 if (fabs(W) > Wex) {
166 W = edgSign(type) * Wex;
167 }
168
169 // DEBUG_SYNAPSE(
170 // printf("AllSTDPSynapses::stdpLearning:\n");
171 // printf(" iEdg: %d\n", iEdg);
172 // printf(" delta: %f\n", delta);
173 // printf(" epre: %f\n", epre);
174 // printf(" epost: %f\n", epost);
175 // printf(" dw: %f\n", dw);
176 // printf(" W: %f\n\n", W);
177 // );
178}

235 int idx = blockIdx.x * blockDim.x + threadIdx.x;
236 if ( idx >= totalSynapseCount )
237 return;
238
239
240 BGSIZE iEdg = edgeIndexMapDevice->incomingEdgeIndexMap_[idx];
241
242 BGFLOAT &psr = allEdgesDevice->psr_[iEdg];
243 BGFLOAT decay = allEdgesDevice->decay_[iEdg];
244
245 // Checks if there is an input spike in the queue.
246 bool isFired = isSpikingSynapsesSpikeQueueDevice(allEdgesDevice, iEdg);
247
248 // is an input in the queue?
249 if (isFired) {
250 switch (classSynapses_d) {
251 case classAllSpikingSynapses:
252 changeSpikingSynapsesPSRDevice(static_cast<AllSpikingSynapsesDeviceProperties*>(allEdgesDevice), iEdg, simulationStep, deltaT);
253 break;
254 case classAllDSSynapses:
255 changeDSSynapsePSRDevice(static_cast<AllDSSynapsesDeviceProperties*>(allEdgesDevice), iEdg, simulationStep, deltaT);
256 break;
257 default:
258 assert(false);
259 }
260 }
261 // decay the post spike response
262 psr *= decay;
263}

276__global__ void advanceSTDPSynapsesDevice ( int totalSynapseCount, EdgeIndexMap* edgeIndexMapDevice, uint64_t simulationStep, const BGFLOAT deltaT, AllSTDPSynapsesDeviceProperties* allEdgesDevice, AllSpikingNeuronsDeviceProperties* allVerticesDevice, int maxSpikes ) {

278 if ( idx >= totalSynapseCount )
279 return;
280
281 BGSIZE iEdg = edgeIndexMapDevice->incomingEdgeIndexMap_[idx];
282
283 // If the synapse is inhibitory or its weight is zero, update synapse state using AllSpikingSynapses::advanceEdge method
284 BGFLOAT &W = allEdgesDevice->W_[iEdg];
285 if(W <= 0.0) {
286 BGFLOAT &psr = allEdgesDevice->psr_[iEdg];
287 BGFLOAT decay = allEdgesDevice->decay_[iEdg];
288

291

293 if (isFired) {
294 switch (classSynapses_d) {
295 case classAllSTDPSynapses:
296 changeSpikingSynapsesPSRDevice(static_cast<AllSpikingSynapsesDeviceProperties*>(allEdgesDevice), iEdg, simulationStep, deltaT);
297 break;
298 case classAllDynamicSTDPSynapses:
299 // Note: we cast void * over the allEdgesDevice, then recast it,
300 // because AllDSSynapsesDeviceProperties inherited properties from
301 // the AllDSSynapsesDeviceProperties and the AllSTDPSynapsesDeviceProperties.
302 changeDSSynapsePSRDevice(static_cast<AllDSSynapsesDeviceProperties*>((void *)allEdgesDevice), iEdg, simulationStep, deltaT);
303 break;
304 default:
305 assert(false);
306 }
307 }
308 // decay the post spike response
309 psr *= decay;
310 return;
311 }
312
313 BGFLOAT &decay = allEdgesDevice->decay_[iEdg];
314 BGFLOAT &psr = allEdgesDevice->psr_[iEdg];
315
316 // is an input in the queue?
317 bool fPre = isSpikingSynapsesSpikeQueueDevice(allEdgesDevice, iEdg);
318 bool fPost = isSTDPSynapseSpikeQueuePostDevice(allEdgesDevice, iEdg);
319 if (fPre || fPost) {
320 BGFLOAT &tauspre = allEdgesDevice->tauspre_[iEdg];
321 BGFLOAT &tauspost = allEdgesDevice->tauspost_[iEdg];
322 BGFLOAT &taupos = allEdgesDevice->taupos_[iEdg];
323 BGFLOAT &tauneg = allEdgesDevice->tauneg_[iEdg];
324 int &totalDelay = allEdgesDevice->totalDelay_[iEdg];
325 bool &useFroemkeDanSTDP = allEdgesDevice->useFroemkeDanSTDP_[iEdg];
326
327 // pre and post neurons index
328 int idxPre = allEdgesDevice->sourceVertexIndex_[iEdg];
329 int idxPost = allEdgesDevice->destVertexIndex_[iEdg];
330 int64_t spikeHistory, spikeHistory2;
331 BGFLOAT delta;
332 BGFLOAT epre, epost;
333
334 if (fPre) { // preSpikeHit
335 // spikeCount points to the next available position of spike_history,
336 // so the getSpikeHistory w/offset = -2 will return the spike time
337 // just one before the last spike.
338 spikeHistory = getSTDPSynapseSpikeHistoryDevice(allVerticesDevice, idxPre, -2, maxSpikes);
339 if (spikeHistory > 0 && useFroemkeDanSTDP) {
340 // delta will include the transmission delay
341 delta = static_cast<BGFLOAT>(simulationStep - spikeHistory) * deltaT;
342 epre = 1.0 - exp(-delta / tauspre);
343 } else {
344 epre = 1.0;
345 }
346
347 // call the learning function stdpLearning() for each pair of
348 // pre-post spikes
349 int offIndex = -1; // last spike
350 while (true) {
351 spikeHistory = getSTDPSynapseSpikeHistoryDevice(allVerticesDevice, idxPost, offIndex, maxSpikes);
352 if (spikeHistory == ULONG_MAX)
353 break;
354 // delta is the spike interval between pre-post spikes
355 delta = -static_cast<BGFLOAT>(simulationStep - spikeHistory) * deltaT;
356
357 if (delta <= -3.0 * tauneg)
358 break;
359 if (useFroemkeDanSTDP) {
360 spikeHistory2 = getSTDPSynapseSpikeHistoryDevice(allVerticesDevice, idxPost, offIndex-1, maxSpikes);
361 if (spikeHistory2 == ULONG_MAX)
362 break;
363 epost = 1.0 - exp(-(static_cast<BGFLOAT>(spikeHistory - spikeHistory2) * deltaT) / tauspost);
364 } else {
365 epost = 1.0;
366 }
367 stdpLearningDevice(allEdgesDevice, iEdg, delta, epost, epre);
368 --offIndex;
369 }
370
371 switch (classSynapses_d) {
372 case classAllSTDPSynapses:
373 changeSpikingSynapsesPSRDevice(static_cast<AllSpikingSynapsesDeviceProperties*>(allEdgesDevice), iEdg, simulationStep, deltaT);
374 break;
375 case classAllDynamicSTDPSynapses:
376 // Note: we cast void * over the allEdgesDevice, then recast it,
377 // because AllDSSynapsesDeviceProperties inherited properties from
378 // the AllDSSynapsesDeviceProperties and the AllSTDPSynapsesDeviceProperties.
379 changeDSSynapsePSRDevice(static_cast<AllDSSynapsesDeviceProperties*>((void *)allEdgesDevice), iEdg, simulationStep, deltaT);
380 break;
381 default:
382 assert(false);
383 }
384 }
385
386 if (fPost) { // postSpikeHit
387 // spikeCount points to the next available position of spike_history,
388 // so the getSpikeHistory w/offset = -2 will return the spike time
389 // just one before the last spike.
390 spikeHistory = getSTDPSynapseSpikeHistoryDevice(allVerticesDevice, idxPost, -2, maxSpikes);
391 if (spikeHistory > 0 && useFroemkeDanSTDP) {
392 // delta will include the transmission delay
393 delta = static_cast<BGFLOAT>(simulationStep - spikeHistory) * deltaT;
394 epost = 1.0 - exp(-delta / tauspost);
395 } else {
396 epost = 1.0;
397 }
398
399 // call the learning function stdpLearning() for each pair of
400 // post-pre spikes
401 int offIndex = -1; // last spike
402 while (true) {
403 spikeHistory = getSTDPSynapseSpikeHistoryDevice(allVerticesDevice, idxPre, offIndex, maxSpikes);
404 if (spikeHistory == ULONG_MAX)
405 break;
406
407 if(spikeHistory + totalDelay > simulationStep) {
408 --offIndex;
409 continue;
410 }
411 // delta is the spike interval between post-pre spikes
412 delta = static_cast<BGFLOAT>(simulationStep - spikeHistory - totalDelay) * deltaT;
413
414
415 if (delta >= 3.0 * taupos)
416 break;
417 if (useFroemkeDanSTDP) {
418 spikeHistory2 = getSTDPSynapseSpikeHistoryDevice(allVerticesDevice, idxPre, offIndex-1, maxSpikes);
419 if (spikeHistory2 == ULONG_MAX)
420 break;
421 epre = 1.0 - exp(-(static_cast<BGFLOAT>(spikeHistory - spikeHistory2) * deltaT) / tauspre);
422 } else {
423 epre = 1.0;
424 }
425 stdpLearningDevice(allEdgesDevice, iEdg, delta, epost, epre);
426 --offIndex;
427 }
428 }
429 }
430
431 // decay the post spike response
432 psr *= decay;
433}

452__device__ void createSpikingSynapse(AllSpikingSynapsesDeviceProperties* allEdgesDevice, const int neuronIndex, const int synapseOffset, int sourceIndex, int destIndex, BGFLOAT *sumPoint, const BGFLOAT deltaT, edgeType type)

468 allEdgesDevice->type_[iEdg] = type;
469
470 allEdgesDevice->tau_[iEdg] = DEFAULT_tau;
471
472 BGFLOAT tau;
473 switch (type) {
474 case II:
475 tau = 6e-3;
476 delay = 0.8e-3;
477 break;
478 case IE:
479 tau = 6e-3;
480 delay = 0.8e-3;
481 break;
482 case EI:
483 tau = 3e-3;
484 delay = 0.8e-3;

874__global__ void updateSynapsesWeightsDevice( int numVertices, BGFLOAT deltaT, BGFLOAT* W_d, int maxEdges, AllSpikingNeuronsDeviceProperties* allVerticesDevice, AllSpikingSynapsesDeviceProperties* allEdgesDevice, vertexType* neuronTypeMap_d )
875{
876 int idx = blockIdx.x * blockDim.x + threadIdx.x;
877 if ( idx >= numVertices )
878 return;
879
880 int adjusted = 0;
881 //int could_have_been_removed = 0; // TODO: use this value
882 int removed = 0;
883 int added = 0;
884
885 // Scale and add sign to the areas
886 // visit each neuron 'a'
887 int destVertex = idx;
888
889 // and each destination neuron 'b'
890 for (int srcVertex = 0; srcVertex < numVertices; srcVertex++) {
891 // visit each synapse at (xa,ya)
892 bool connected = false;
893 edgeType type = edgType(neuronTypeMap_d, srcVertex, destVertex);
894
895 // for each existing synapse
896 BGSIZE existing_synapses = allEdgesDevice->edgeCounts_[destVertex];
897 int existingSynapsesChecked = 0;
898 for (BGSIZE synapseIndex = 0; (existingSynapsesChecked < existing_synapses) && !connected; synapseIndex++) {
899 BGSIZE iEdg = maxEdges * destVertex + synapseIndex;
900 if (allEdgesDevice->inUse_[iEdg] == true) {
901 // if there is a synapse between a and b
902 if (allEdgesDevice->sourceVertexIndex_[iEdg] == srcVertex) {
903 connected = true;
904 adjusted++;
905
906 // adjust the strength of the synapse or remove
907 // it from the synapse map if it has gone below
908 // zero.
909 if (W_d[srcVertex * numVertices + destVertex] <= 0) {
910 removed++;
911 eraseSpikingSynapse(allEdgesDevice, destVertex, synapseIndex, maxEdges);
912 } else {
913 // adjust
914 // g_synapseStrengthAdjustmentConstant is 1.0e-8;
915 allEdgesDevice->W_[iEdg] = W_d[srcVertex * numVertices
916 + destVertex] * edgSign(type) * AllNeuroEdges::SYNAPSE_STRENGTH_ADJUSTMENT;
917 }
918 }
919 existingSynapsesChecked++;
920 }
921 }
922
923 // if not connected and weight(a,b) > 0, add a new synapse from a to b
924 if (!connected && (W_d[srcVertex * numVertices + destVertex] > 0)) {
925 // locate summation point
926 BGFLOAT* sumPoint = &( allVerticesDevice->summationMap_[destVertex] );
927 added++;
928
929 addSpikingSynapse(allEdgesDevice, type, srcVertex, destVertex, srcVertex, destVertex, sumPoint, deltaT, W_d, numVertices);
930 }
931 }
932}
