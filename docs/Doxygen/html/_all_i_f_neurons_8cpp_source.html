181void AllIFNeurons::createNeuron(int i, Layout *layout)
182{
183 // set the neuron info for neurons
184 Iinject_[i] = initRNG.inRange(IinjectRange_[0], IinjectRange_[1]);
185 Inoise_[i] = initRNG.inRange(InoiseRange_[0], InoiseRange_[1]);
186 Vthresh_[i] = initRNG.inRange(VthreshRange_[0], VthreshRange_[1]);
187 Vrest_[i] = initRNG.inRange(VrestingRange_[0], VrestingRange_[1]);
188 Vreset_[i] = initRNG.inRange(VresetRange_[0], VresetRange_[1]);
189 Vinit_[i] = initRNG.inRange(VinitRange_[0], VinitRange_[1]);
190 Vm_[i] = Vinit_[i];
191
192 initNeuronConstsFromParamValues(i, Simulator::getInstance().getDeltaT());
193
194 switch (layout->vertexTypeMap_[i]) {
195 case INH:
196 LOG4CPLUS_DEBUG(vertexLogger_, "Setting inhibitory neuron: " << i);
197 // set inhibitory absolute refractory period
198 Trefract_[i] = DEFAULT_InhibTrefract; // TODO(derek): move defaults inside model.
199 break;
200
201 case EXC:
202 LOG4CPLUS_DEBUG(vertexLogger_, "Setting excitatory neuron: " << i);
203 // set excitatory absolute refractory period
204 Trefract_[i] = DEFAULT_ExcitTrefract;
205 break;
206
207 default:
208 LOG4CPLUS_DEBUG(vertexLogger_,
209 "ERROR: unknown neuron type: " << layout->vertexTypeMap_[i] << "@" << i);
210 assert(false);
211 break;
212 }
213 // endogenously_active_neuron_map -> Model State
214 if (layout->starterMap_[i]) {
215 // set endogenously active threshold voltage, reset voltage, and refractory period
216 Vthresh_[i] = initRNG.inRange(starterVthreshRange_[0], starterVthreshRange_[1]);
217 Vreset_[i] = initRNG.inRange(starterVresetRange_[0], starterVresetRange_[1]);
218 Trefract_[i] = DEFAULT_ExcitTrefract; // TODO(derek): move defaults inside model.
219 }
220
221 LOG4CPLUS_DEBUG(vertexLogger_, "\nCREATE NEURON[" << i << "] {" << endl
222 << "\tVm = " << Vm_[i] << endl
223 << "\tVthresh = " << Vthresh_[i] << endl
224 << "\tI0 = " << I0_[i] << endl
225 << "\tInoise = " << Inoise_[i] << " from : ("
226 << InoiseRange_[0] << "," << InoiseRange_[1]
227 << ")" << endl
228 << "\tC1 = " << C1_[i] << endl
229 << "\tC2 = " << C2_[i] << endl
230 << "}" << endl);
231}

254void AllIFNeurons::initNeuronConstsFromParamValues(int i, const BGFLOAT deltaT)
255{
256 BGFLOAT &Tau = this->Tau_[i];
257 BGFLOAT &C1 = this->C1_[i];
258 BGFLOAT &C2 = this->C2_[i];
259 BGFLOAT &Rm = this->Rm_[i];
260 BGFLOAT &I0 = this->I0_[i];
261 BGFLOAT &Iinject = this->Iinject_[i];
262 BGFLOAT &Vrest = this->Vrest_[i];
263
264 /* init consts C1,C2 for exponential Euler integration */
265 if (Tau > 0) {
266 C1 = exp(-deltaT / Tau);
267 C2 = Rm * (1 - C1);
268 } else {
269 C1 = 0.0;
270 C2 = Rm;
271 }
272 /* calculate const IO */
273 if (Rm > 0) {
274 I0 = Iinject + Vrest / Rm;
275 } else {
276 assert(false);
277 }
278}

284string AllIFNeurons::toString(const int index) const

297 ss << "nStepsInRefr: " << numStepsInRefractoryPeriod_[index]
