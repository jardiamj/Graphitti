78void Layout::setupLayout()
79{
80 int numVertices = Simulator::getInstance().getTotalVertices();
81
82 // Allocate memory
83 xloc_ = new VectorMatrix(MATRIX_TYPE, MATRIX_INIT, 1, numVertices);
84 yloc_ = new VectorMatrix(MATRIX_TYPE, MATRIX_INIT, 1, numVertices);
85 dist2_ = new CompleteMatrix(MATRIX_TYPE, MATRIX_INIT, numVertices, numVertices);
86 dist_ = new CompleteMatrix(MATRIX_TYPE, MATRIX_INIT, numVertices, numVertices);
87
88 // Initialize neuron locations memory, grab global info
89 initVerticesLocs();
90
91 // computing distance between each pair of vertices given each vertex's xy location
92 for (int n = 0; n < numVertices - 1; n++) {
93 for (int n2 = n + 1; n2 < numVertices; n2++) {
94 // distance^2 between two points in point-slope form
95 (*dist2_)(n, n2) = ((*xloc_)[n] - (*xloc_)[n2]) * ((*xloc_)[n] - (*xloc_)[n2])
96 + ((*yloc_)[n] - (*yloc_)[n2]) * ((*yloc_)[n] - (*yloc_)[n2]);
97
98 // both points are equidistant from each other
99 (*dist2_)(n2, n) = (*dist2_)(n, n2);
100 }
101 }
102
103 // take the square root to get actual distance (Pythagoras was right!)
104 (*dist_) = sqrt((*dist2_));
105
106 // more allocation of internal memory
107 vertexTypeMap_ = new vertexType[numVertices]; // todo: make array into vector
108 starterMap_ = new bool[numVertices]; // todo: make array into vector
109}

154void Layout::initVerticesLocs()
155{
156 int numVertices = Simulator::getInstance().getTotalVertices();
157
158 // Initialize vertex locations
159 if (gridLayout_) {
160 // grid layout
161 for (int i = 0; i < numVertices; i++) {
162 (*xloc_)[i] = i % Simulator::getInstance().getHeight();
163 (*yloc_)[i] = i / Simulator::getInstance().getHeight();
164 }
165 } else {
166 // random layout
167 for (int i = 0; i < numVertices; i++) {
168 (*xloc_)[i] = initRNG.inRange(0, Simulator::getInstance().getWidth());
169 (*yloc_)[i] = initRNG.inRange(0, Simulator::getInstance().getHeight());
170 }
171 }
