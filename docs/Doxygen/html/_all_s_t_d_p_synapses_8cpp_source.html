313void AllSTDPSynapses::advanceEdge(const BGSIZE iEdg, AllVertices *neurons)
314{
315 // If the synapse is inhibitory or its weight is zero, update synapse state using AllSpikingSynapses::advanceEdge method
316 //LOG4CPLUS_DEBUG(edgeLogger_, "iEdg : " << iEdg );
317
318 BGFLOAT &W = W_[iEdg];
319 /*
320 if (W <= 0.0) {
321 AllSpikingSynapses::advanceEdge(iEdg, neurons);
322 return;
323 }
324 */
325
326 BGFLOAT &decay = decay_[iEdg];
327 BGFLOAT &psr = psr_[iEdg];
328 BGFLOAT &summationPoint = *(summationPoint_[iEdg]);
329
330 // is an input in the queue?
331 bool fPre = isSpikeQueue(iEdg);
332 bool fPost = isSpikeQueuePost(iEdg);
333
334 if (fPre || fPost) {
335 const BGFLOAT taupos = taupos_[iEdg];
336 const BGFLOAT tauneg = tauneg_[iEdg];
337 const int total_delay = totalDelay_[iEdg];
338
339 BGFLOAT deltaT = Simulator::getInstance().getDeltaT();
340 AllSpikingNeurons *spNeurons = dynamic_cast<AllSpikingNeurons *>(neurons);
341
342 // pre and post neurons index
343 int idxPre = sourceVertexIndex_[iEdg];
344 int idxPost = destVertexIndex_[iEdg];
345 uint64_t spikeHistory, spikeHistory2;
346 BGFLOAT delta;
347 BGFLOAT epre, epost;
348
349 if (fPre) { // preSpikeHit
350 // spikeCount points to the next available position of spike_history,
351 // so the getSpikeHistory w/offset = -2 will return the spike time
352 // just one before the last spike.
353 spikeHistory = spNeurons->getSpikeHistory(idxPre, -2);
354
355 epre = 1.0;
356 epost = 1.0;
357 // call the learning function stdpLearning() for each pair of
358 // pre-post spikes
359 int offIndex = -1; // last spike
360 while (true) {
361 spikeHistory = spNeurons->getSpikeHistory(idxPost, offIndex);
362 if (spikeHistory == ULONG_MAX)
363 break;
364 // delta is the spike interval between pre-post spikes
365 // (include pre-synaptic transmission delay)
366 delta = -static_cast<BGFLOAT>(g_simulationStep - spikeHistory) * deltaT;
367 /*
368 LOG4CPLUS_DEBUG(fileLogger_,"\nAllSTDPSynapses::advanceSynapse: fPre" << endl
369 << "\tiEdg: " << iEdg << endl
370 << "\tidxPre: " << idxPre << endl
371 << "\tidxPost: " << idxPost << endl
372 << "\tspikeHistory: " << spikeHistory << endl
373 << "\tepre: " << epre << endl
374 << "\tepost: " << epost << endl
375 << "\tg_simulationStep: " << g_simulationStep << endl
376 << "\tdelta: " << delta << endl << endl);
377 */
378
379 if (delta <= -3.0 * tauneg)
380 break;
381
382 stdpLearning(iEdg, delta, epost, epre, idxPre, idxPost);
383 --offIndex;
384 }
385
386 changePSR(iEdg, deltaT);
387 }
388
389 if (fPost) { // postSpikeHit
390 // spikeCount points to the next available position of spike_history,
391 // so the getSpikeHistory w/offset = -2 will return the spike time
392 // just one before the last spike.
393 spikeHistory = spNeurons->getSpikeHistory(idxPost, -2);
394 epost = 1.0;
395 epre = 1;
396
397
398 // call the learning function stdpLearning() for each pair of
399 // post-pre spikes
400 int offIndex = -1; // last spike
401 while (true) {
402 spikeHistory = spNeurons->getSpikeHistory(idxPre, offIndex);
403 if (spikeHistory == ULONG_MAX)
404 break;
405
406 if (spikeHistory + total_delay > g_simulationStep) {
407 --offIndex;
408 continue;
409 }
410 // delta is the spike interval between post-pre spikes
411 delta = static_cast<BGFLOAT>(g_simulationStep - spikeHistory - total_delay) * deltaT;
412 /*
413 LOG4CPLUS_DEBUG(fileLogger_,"\nAllSTDPSynapses::advanceSynapse: fPost" << endl
414 << "\tiEdg: " << iEdg << endl
415 << "\tidxPre: " << idxPre << endl
416 << "\tidxPost: " << idxPost << endl
417 << "\tspikeHistory: " << spikeHistory << endl
418 << "\tg_simulationStep: " << g_simulationStep << endl
419 << "\tepre: " << epre << endl
420 << "\tepost: " << epost << endl
421 << "\tdelta: " << delta << endl << endl);
422 */
423 if (delta >= 3.0 * taupos)
424 break;
425
426 stdpLearning(iEdg, delta, epost, epre, idxPre, idxPost);
427 --offIndex;
428 }
429 }
430 }
431
432 // decay the post spike response
433 psr *= decay;
434 // and apply it to the summation point
435 #ifdef USE_OMP
436 #pragma omp atomic
437 #endif
438 summationPoint += psr;
439 #ifdef USE_OMP
440 //PAB: atomic above has implied flush (following statement generates error -- can't be member variable)
441 //#pragma omp flush (summationPoint)
442 #endif
443}

446BGFLOAT AllSTDPSynapses::synapticWeightModification(const BGSIZE iEdg, BGFLOAT synapticWeight,
447 double delta)
448{
449 BGFLOAT STDPgap = STDPgap_[iEdg];
450 BGFLOAT muneg = muneg_[iEdg];
451 BGFLOAT mupos = mupos_[iEdg];
452 BGFLOAT tauneg = tauneg_[iEdg];
453 BGFLOAT taupos = taupos_[iEdg];
454 BGFLOAT Aneg = Aneg_[iEdg];
455 BGFLOAT Apos = Apos_[iEdg];
456 BGFLOAT Wex = Wex_[iEdg];
457 BGFLOAT &W = W_[iEdg];
458 edgeType type = type_[iEdg];
459 BGFLOAT dw = 0;
460 BGFLOAT oldW = W;
461
462 // BGFLOAT modDelta = fabs(delta);
463
464 if (delta < -STDPgap) {
465 // depression
466
467 dw = pow(fabs(W) / Wex, muneg) * Aneg * exp(delta / tauneg); // normalize
468 } else if (delta > STDPgap) {
469 // potentiation
470 dw = pow(fabs(Wex - fabs(W)) / Wex, mupos) * Apos * exp(-delta / taupos); // normalize
471 }
472
473 return dw;
474}

486void AllSTDPSynapses::stdpLearning(const BGSIZE iEdg, double delta, double epost, double epre,
487 int srcVertex, int destVertex)
488{
489 BGFLOAT STDPgap = STDPgap_[iEdg];
490 BGFLOAT muneg = muneg_[iEdg];
491 BGFLOAT mupos = mupos_[iEdg];
492 BGFLOAT tauneg = tauneg_[iEdg];
493 BGFLOAT taupos = taupos_[iEdg];
494 BGFLOAT Aneg = Aneg_[iEdg];
495 BGFLOAT Apos = Apos_[iEdg];
496 BGFLOAT Wex = Wex_[iEdg];
497 BGFLOAT &W = W_[iEdg];
498 edgeType type = type_[iEdg];
499 BGFLOAT oldW = W;
500 // BGFLOAT modDelta = fabs(delta);
501
502 if (delta <= fabs(STDPgap)) {
503 return;
504 }
505
506 // dw is the fractional change in synaptic strength; add 1.0 to become the scaling ratio
507 //dw = 1.0 + dw * epre * epost;
508 BGFLOAT dw = 1.0 + synapticWeightModification(iEdg, W, delta);
509
510 // if scaling ratio is less than zero, set it to zero so this synapse, its
511 // strength is always zero
512 // TODO: Where is the code for this?
513
514 // current weight multiplies dw (scaling ratio) to generate new weight
515 if (dw != 0.0)
516 W *= dw;
517
518 // if new weight is bigger than Wex (maximum allowed weight), then set it to Wex
519 if (fabs(W) > Wex) {
520 W = edgSign(type) * Wex;
521 }
522 /*

524 "iEdg value " << iEdg
525 << "; source:" << srcVertex
526 << "; dest:" << destVertex
527 << "; delta:" << delta
528 << "; oldW:" << oldW

540{
541 uint32_t &delayQueue = delayQueuePost_[iEdg];

556void AllSTDPSynapses::postSpikeHit(const BGSIZE iEdg)
557{
558 uint32_t &delay_queue = delayQueuePost_[iEdg];
559 int &delayIdx = delayIndexPost_[iEdg];
560 int &ldelayQueue = delayQueuePostLength_[iEdg];
561 int &total_delay = totalDelayPost_[iEdg];
562
563 // Add to spike queue
564
565 // calculate index where to insert the spike into delayQueue
566 int idx = delayIdx + total_delay;
567 if (idx >= ldelayQueue) {
568 idx -= ldelayQueue;
569 }
570
571 // set a spike
572 assert(!(delay_queue & (0x1 << idx)));
573 delay_queue |= (0x1 << idx);
574}
