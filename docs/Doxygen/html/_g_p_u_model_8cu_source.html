41void GPUModel::allocDeviceStruct(void** allVerticesDevice, void** allEdgesDevice)
42{
43 // Get neurons and synapses
44 shared_ptr<AllVertices> neurons = layout_->getVertices();
45 shared_ptr<AllEdges> synapses = connections_->getEdges();
46
47 // Allocate Neurons and Synapses structs on GPU device memory
48 neurons->allocNeuronDeviceStruct(allVerticesDevice);
49 synapses->allocEdgeDeviceStruct(allEdgesDevice);
50
51 // Allocate memory for random noise array
52 int numVertices = Simulator::getInstance().getTotalVertices();
53 BGSIZE randNoise_d_size = numVertices * sizeof (float); // size of random noise array
54 HANDLE_ERROR( cudaMalloc ( ( void ** ) &randNoise_d, randNoise_d_size ) );
55

57 neurons->copyNeuronHostToDevice( *allVerticesDevice );
58 synapses->copyEdgeHostToDevice( *allEdgesDevice );
59
60 // Allocate synapse inverse map in device memory
61 allocSynapseImap( numVertices );
62}

85void GPUModel::setupSim()
86{
87 // Set device ID
88 HANDLE_ERROR( cudaSetDevice( g_deviceId ) );
89 // Set DEBUG flag
90 HANDLE_ERROR( cudaMemcpyToSymbol (d_debug_mask, &g_debug_mask, sizeof(int) ) );
91 Model::setupSim();
92
93 //initialize Mersenne Twister
94 //assuming numVertices >= 100 and is a multiple of 100. Note rng_mt_rng_count must be <= MT_RNG_COUNT
95 int rng_blocks = 25; //# of blocks the kernel will use
96 int rng_nPerRng = 4; //# of iterations per thread (thread granularity, # of rands generated per thread)
97 int rng_mt_rng_count = Simulator::getInstance().getTotalVertices() / rng_nPerRng; //# of threads to generate for numVertices rand #s
98 int rng_threads = rng_mt_rng_count/rng_blocks; //# threads per block needed
99 initMTGPU(Simulator::getInstance().getNoiseRngSeed(), rng_blocks, rng_threads, rng_nPerRng, rng_mt_rng_count);
100
101#ifdef PERFORMANCE_METRICS
102 cudaEventCreate( &start );
103 cudaEventCreate( &stop );
104
105 t_gpu_rndGeneration = 0.0;
106 t_gpu_advanceNeurons = 0.0;
107 t_gpu_advanceSynapses = 0.0;
108 t_gpu_calcSummation = 0.0;
109#endif // PERFORMANCE_METRICS
110
111 // allocates memories on CUDA device
112 allocDeviceStruct((void **)&allVerticesDevice_, (void **)&allEdgesDevice_);
113
114 // copy inverse map to the device memory
115 copySynapseIndexMapHostToDevice(*(connections_->getEdgeIndexMap().get()), Simulator::getInstance().getTotalVertices());
116

140void GPUModel::advance()
141{
142#ifdef PERFORMANCE_METRICS
143 // Reset CUDA timer to start measurement of GPU operations
144 cudaStartTimer();
145#endif // PERFORMANCE_METRICS
146
147 // Get neurons and synapses
148 shared_ptr<AllVertices> neurons = layout_->getVertices();
149 shared_ptr<AllEdges> synapses = connections_->getEdges();
150
151 normalMTGPU(randNoise_d);
152
153#ifdef PERFORMANCE_METRICS
154 cudaLapTime(t_gpu_rndGeneration);
155 cudaStartTimer();
156#endif // PERFORMANCE_METRICS
157
158 // display running info to console
159 // Advance neurons ------------->
160 dynamic_cast<AllSpikingNeurons *>(neurons.get())->advanceVertices(*(connections_->getEdges().get()), allVerticesDevice_, allEdgesDevice_, randNoise_d, synapseIndexMapDevice_);
161
162#ifdef PERFORMANCE_METRICS
163 cudaLapTime(t_gpu_advanceNeurons);
164 cudaStartTimer();
165#endif // PERFORMANCE_METRICS
166
167 // Advance synapses ------------->
168 synapses->advanceEdges(allEdgesDevice_, allVerticesDevice_, synapseIndexMapDevice_);
169
170#ifdef PERFORMANCE_METRICS
171 cudaLapTime(t_gpu_advanceSynapses);
172 cudaStartTimer();
173#endif // PERFORMANCE_METRICS
174
175 // calculate summation point
176 calcSummationMap();
177
178#ifdef PERFORMANCE_METRICS
179 cudaLapTime(t_gpu_calcSummation);
180#endif // PERFORMANCE_METRICS

266void GPUModel::copySynapseIndexMapHostToDevice(EdgeIndexMap &synapseIndexMapHost, int numVertices)
267{
268 shared_ptr<AllEdges> synapses = connections_->getEdges();
269 int totalSynapseCount = dynamic_cast<AllEdges*>(synapses.get())->totalEdgeCount_;
270
271 if (totalSynapseCount == 0)
272 return;
273
274 // TODO: rename variable, DevicePointer
275 EdgeIndexMap synapseIMapDevice;
276
277 HANDLE_ERROR( cudaMemcpy ( &synapseIMapDevice, synapseIndexMapDevice_,
278 sizeof( EdgeIndexMap ), cudaMemcpyDeviceToHost ) );
279
280 // forward map
281 HANDLE_ERROR( cudaMemcpy ( synapseIMapDevice.outgoingEdgeBegin_,
282 synapseIndexMapHost.outgoingEdgeBegin_, numVertices * sizeof( BGSIZE ), cudaMemcpyHostToDevice ) );
283 HANDLE_ERROR( cudaMemcpy ( synapseIMapDevice.outgoingEdgeCount_,
284 synapseIndexMapHost.outgoingEdgeCount_, numVertices * sizeof( BGSIZE ), cudaMemcpyHostToDevice ) );
285 // the number of synapses may change, so we reallocate the memory
286 if (synapseIMapDevice.outgoingEdgeIndexMap_ != nullptr) {
287 HANDLE_ERROR( cudaFree( synapseIMapDevice.outgoingEdgeIndexMap_ ) );
288 }
289 HANDLE_ERROR( cudaMalloc( ( void ** ) &synapseIMapDevice.outgoingEdgeIndexMap_,
290 totalSynapseCount * sizeof( BGSIZE ) ) );
291 HANDLE_ERROR( cudaMemcpy ( synapseIMapDevice.outgoingEdgeIndexMap_, synapseIndexMapHost.outgoingEdgeIndexMap_,
292 totalSynapseCount * sizeof( BGSIZE ), cudaMemcpyHostToDevice ) );
293
294 // active synapse map
295 HANDLE_ERROR( cudaMemcpy ( synapseIMapDevice.incomingEdgeBegin_,
296 synapseIndexMapHost.incomingEdgeBegin_, numVertices * sizeof( BGSIZE ), cudaMemcpyHostToDevice ) );
297 HANDLE_ERROR( cudaMemcpy ( synapseIMapDevice.incomingEdgeCount_,
298 synapseIndexMapHost.incomingEdgeCount_, numVertices * sizeof( BGSIZE ), cudaMemcpyHostToDevice ) );
299 // the number of synapses may change, so we reallocate the memory
300 if (synapseIMapDevice.incomingEdgeIndexMap_ != nullptr) {
301 HANDLE_ERROR( cudaFree( synapseIMapDevice.incomingEdgeIndexMap_ ) );
302 }
303 HANDLE_ERROR( cudaMalloc( ( void ** ) &synapseIMapDevice.incomingEdgeIndexMap_,
304 totalSynapseCount * sizeof( BGSIZE ) ) );
305 HANDLE_ERROR( cudaMemcpy ( synapseIMapDevice.incomingEdgeIndexMap_, synapseIndexMapHost.incomingEdgeIndexMap_,
306 totalSynapseCount * sizeof( BGSIZE ), cudaMemcpyHostToDevice ) );
307
308 HANDLE_ERROR( cudaMemcpy ( synapseIndexMapDevice_, &synapseIMapDevice,
309 sizeof( EdgeIndexMap ), cudaMemcpyHostToDevice ) );
310}

328__global__ void calcSummationMapDevice(int totalVertices,
329 AllSpikingNeuronsDeviceProperties* __restrict__ allVerticesDevice,
330 const EdgeIndexMap* __restrict__ synapseIndexMapDevice_,
331 const AllSpikingSynapsesDeviceProperties* __restrict__ allEdgesDevice)
332{
333 // The usual thread ID calculation and guard against excess threads
334 // (beyond the number of vertices, in this case).
335 int idx = blockIdx.x * blockDim.x + threadIdx.x;
336 if ( idx >= totalVertices )
337 return;
338
339 // Number of incoming synapses
340 const BGSIZE synCount = synapseIndexMapDevice_->incomingEdgeCount_[idx];
341 // Optimization: terminate thread if no incoming synapses
342 if (synCount != 0) {
343 // Index of start of this neuron's block of forward map entries
344 const int beginIndex = synapseIndexMapDevice_->incomingEdgeBegin_[idx];
345 // Address of the start of this neuron's block of forward map entries
346 const BGSIZE* activeMapBegin =
347 &(synapseIndexMapDevice_->incomingEdgeIndexMap_[beginIndex]);
348 // Summed post-synaptic response (PSR)
349 BGFLOAT sum = 0.0;
350 // Index of the current incoming synapse
351 BGSIZE synIndex;
352 // Repeat for each incoming synapse
353 for (BGSIZE i = 0; i < synCount; i++) {
354 // Get index of current incoming synapse
355 synIndex = activeMapBegin[i];
356 // Fetch its PSR and add into sum
357 sum += allEdgesDevice->psr_[synIndex];
358 }
359 // Store summed PSR into this neuron's summation point
360 allVerticesDevice->summationMap_[idx] = sum;
361 }
362}
