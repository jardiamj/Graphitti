33void AllLIFNeurons::advanceNeuron(const int index)
34{
35 BGFLOAT &Vm = this->Vm_[index];
36 BGFLOAT &Vthresh = this->Vthresh_[index];
37 BGFLOAT &summationPoint = this->summationMap_[index];
38 BGFLOAT &I0 = this->I0_[index];
39 BGFLOAT &Inoise = this->Inoise_[index];
40 BGFLOAT &C1 = this->C1_[index];
41 BGFLOAT &C2 = this->C2_[index];
42 int &nStepsInRefr = this->numStepsInRefractoryPeriod_[index];
43
44 if (nStepsInRefr > 0) {
45 // is neuron refractory?
46 --nStepsInRefr;
47 } else if (Vm >= Vthresh) {
48 // should it fire?
49 fire(index);
50 } else {
51 summationPoint += I0; // add IO
52 // add noise
53 BGFLOAT noise = (*noiseRNG)();
54 //LOG4CPLUS_DEBUG(vertexLogger_, "ADVANCE NEURON[" << index << "] :: Noise = " << noise);
55 summationPoint += noise * Inoise; // add noise
56 Vm = C1 * Vm + C2 * summationPoint; // decay Vm and add inputs
57 }
58 // clear synaptic input for next time step
59 summationPoint = 0;
60
