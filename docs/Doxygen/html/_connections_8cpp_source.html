119void Connections::createSynapsesFromWeights(const int numVertices, Layout *layout,
120 AllVertices &vertices, AllEdges &synapses)
121{
122 // for each neuron
123 for (int i = 0; i < numVertices; i++) {
124 // for each synapse in the vertex
125 for (BGSIZE synapseIndex = 0; synapseIndex < Simulator::getInstance().getMaxEdgesPerVertex();
126 synapseIndex++) {
127 BGSIZE iEdg = Simulator::getInstance().getMaxEdgesPerVertex() * i + synapseIndex;
128 // if the synapse weight is not zero (which means there is a connection), create the synapse
129 if (synapses.W_[iEdg] != 0.0) {
130 BGFLOAT theW = synapses.W_[iEdg];
131 BGFLOAT *sumPoint = &(vertices.summationMap_[i]);
132 int srcVertex = synapses.sourceVertexIndex_[iEdg];
133 int destVertex = synapses.destVertexIndex_[iEdg];
134 edgeType type = layout->edgType(srcVertex, destVertex);
135 synapses.edgeCounts_[i]++;
136 synapses.createEdge(iEdg, srcVertex, destVertex, sumPoint,
137 Simulator::getInstance().getDeltaT(), type);
138 synapses.W_[iEdg] = theW;
139 }
140 }
141 }
