57void ConnStatic::setupConnections(Layout *layout, AllVertices *vertices, AllEdges *edges)
58{
59 Simulator &simulator = Simulator::getInstance();
60 int numVertices = simulator.getTotalVertices();
61 vector<DistDestVertex> distDestVertices[numVertices];
62 BGSIZE maxTotalEdges = simulator.getMaxEdgesPerVertex() * simulator.getTotalVertices();
63 WCurrentEpoch_ = new BGFLOAT[maxTotalEdges];
64 sourceVertexIndexCurrentEpoch_ = new int[maxTotalEdges];
65 destVertexIndexCurrentEpoch_ = new int[maxTotalEdges];
66 AllNeuroEdges *neuroEdges = dynamic_cast<AllNeuroEdges *>(edges);
67
68 radiiSize_ = numVertices;
69 int added = 0;
70
71 LOG4CPLUS_INFO(fileLogger_, "Initializing connections");
72
73 for (int srcVertex = 0; srcVertex < numVertices; srcVertex++) {
74 distDestVertices[srcVertex].clear();
75
76 // pick the connections shorter than threshConnsRadius
77 for (int destVertex = 0; destVertex < numVertices; destVertex++) {
78 if (srcVertex != destVertex) {
79 BGFLOAT dist = (*layout->dist_)(srcVertex, destVertex);
80 if (dist <= threshConnsRadius_) {
81 DistDestVertex distDestVertex {dist, destVertex};
82 distDestVertices[srcVertex].push_back(distDestVertex);
83 }
84 }
85 }
86
87 // sort ascendant
88 sort(distDestVertices[srcVertex].begin(), distDestVertices[srcVertex].end());
89 // pick the shortest connsPerVertex_ connections
90 for (BGSIZE i = 0; i < distDestVertices[srcVertex].size() && (int)i < connsPerVertex_; i++) {
91 int destVertex = distDestVertices[srcVertex][i].destVertex;
92 edgeType type = layout->edgType(srcVertex, destVertex);
93 BGFLOAT *sumPoint = &vertices->summationMap_[destVertex];
94
95 LOG4CPLUS_DEBUG(fileLogger_,
96 "Source: " << srcVertex << " Dest: " << destVertex
97 << " Dist: " << distDestVertices[srcVertex][i].dist);
98
99 BGSIZE iEdg;
100 edges->addEdge(iEdg, type, srcVertex, destVertex, sumPoint, simulator.getDeltaT());
101 added++;
102
103
104 // set edge weight
105 // TODO: we need another synaptic weight distibution mode (normal distribution)
106 if (neuroEdges->edgSign(type) > 0) {
107 neuroEdges->W_[iEdg] = initRNG.inRange(excWeight_[0], excWeight_[1]);
108 } else {
109 neuroEdges->W_[iEdg] = initRNG.inRange(inhWeight_[0], inhWeight_[1]);
110 }
111 }
112 }
113
114 string weight_str = "";
115 for (int i = 0; i < maxTotalEdges; i++) {
116 WCurrentEpoch_[i] = neuroEdges->W_[i];
117 sourceVertexIndexCurrentEpoch_[i] = neuroEdges->sourceVertexIndex_[i];
118 destVertexIndexCurrentEpoch_[i] = neuroEdges->destVertexIndex_[i];
119
120 if (WCurrentEpoch_[i] != 0) {
121 // LOG4CPLUS_DEBUG(edgeLogger_,i << WCurrentEpoch_[i]);
122 weight_str += to_string(WCurrentEpoch_[i]) + " ";
123 }
124 }
125 // LOG4CPLUS_DEBUG(edgeLogger_, "Weights are " << weight_str);
126
127 int nRewiring = added * rewiringProbability_;
128
129 LOG4CPLUS_DEBUG(fileLogger_, "Rewiring connections: " << nRewiring);
130
131 LOG4CPLUS_DEBUG(fileLogger_, "Added connections: " << added);
132}
