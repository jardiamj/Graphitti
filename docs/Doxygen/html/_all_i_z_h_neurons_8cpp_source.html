241void AllIZHNeurons::advanceNeuron(const int index)
242{
243 BGFLOAT &Vm = this->Vm_[index];
244 BGFLOAT &Vthresh = this->Vthresh_[index];
245 BGFLOAT &summationPoint = this->summationMap_[index];
246 BGFLOAT &I0 = this->I0_[index];
247 BGFLOAT &Inoise = this->Inoise_[index];
248 BGFLOAT &C1 = this->C1_[index];
249 BGFLOAT &C2 = this->C2_[index];
250 BGFLOAT &C3 = this->C3_[index];
251 int &nStepsInRefr = this->numStepsInRefractoryPeriod_[index];
252
253 BGFLOAT &a = Aconst_[index];
254 BGFLOAT &b = Bconst_[index];
255 BGFLOAT &u = this->u_[index];
256
257 if (nStepsInRefr > 0) {
258 // is neuron refractory?
259 --nStepsInRefr;
260 } else if (Vm >= Vthresh) {
261 // should it fire?
262 fire(index);
263 } else {
264 summationPoint += I0; // add IO
265 // add noise
266 BGFLOAT noise = (*noiseRNG)();
267 // Happens really often, causes drastic slow down
268 // DEBUG_MID(cout << "ADVANCE NEURON[" << index << "] :: noise = " << noise << endl;)
269 summationPoint += noise * Inoise; // add noise
270
271 BGFLOAT Vint = Vm * 1000;
272
273 // Izhikevich model integration step
274 BGFLOAT Vb = Vint + C3 * (0.04 * Vint * Vint + 5 * Vint + 140 - u);
275 u = u + C3 * a * (b * Vint - u);
276
277 Vm = Vb * 0.001 + C2 * summationPoint; // add inputs
278 }
279
280 // Happens really often, causes drastic slow down
281 // DEBUG_MID(cout << index << " " << Vm << endl;)
282 // DEBUG_MID(cout << "NEURON[" << index << "] {" << endl
283 // << "\tVm = " << Vm << endl
284 // << "\ta = " << a << endl
285 // << "\tb = " << b << endl
286 // << "\tc = " << Cconst_[index] << endl
287 // << "\td = " << Dconst_[index] << endl
288 // << "\tu = " << u << endl
289 // << "\tVthresh = " << Vthresh << endl
290 // << "\tsummationPoint = " << summationPoint << endl
291 // << "\tI0 = " << I0 << endl
292 // << "\tInoise = " << Inoise << endl
293 // << "\tC1 = " << C1 << endl
294 // << "\tC2 = " << C2 << endl
295 // << "\tC3 = " << C3 << endl
296 // << "}" << endl;)
297
298 // clear synaptic input for next time step
299 summationPoint = 0;
300}

305void AllIZHNeurons::fire(const int index)
306{
307 const BGFLOAT deltaT = Simulator::getInstance().getDeltaT();
308 AllSpikingNeurons::fire(index);
309
310 // calculate the number of steps in the absolute refractory period
311 BGFLOAT &Vm = this->Vm_[index];
312 int &nStepsInRefr = this->numStepsInRefractoryPeriod_[index];
313 BGFLOAT &Trefract = this->Trefract_[index];
314
315 BGFLOAT &c = Cconst_[index];
316 BGFLOAT &d = Dconst_[index];
317 BGFLOAT &u = this->u_[index];
318
319 nStepsInRefr = static_cast<int>(Trefract / deltaT + 0.5);
320
321 // reset to 'Vreset'
322 Vm = c * 0.001;
323 u = u + d;
324}
