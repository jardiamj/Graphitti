197void AllSpikingSynapses::createEdge(const BGSIZE iEdg, int srcVertex, int destVertex,
198 BGFLOAT *sumPoint, const BGFLOAT deltaT, edgeType type)
199{
200 BGFLOAT delay;
201
202 inUse_[iEdg] = true;
203 summationPoint_[iEdg] = sumPoint;
204 destVertexIndex_[iEdg] = destVertex;
205 sourceVertexIndex_[iEdg] = srcVertex;
206 W_[iEdg] = edgSign(type) * 10.0e-9;
207 type_[iEdg] = type;
208 tau_[iEdg] = DEFAULT_tau;
209
210 BGFLOAT tau;
211 switch (type) {
212 case II:
213 tau = 6e-3;
214 delay = 0.8e-3;
215 break;
216 case IE:
217 tau = 6e-3;
218 delay = 0.8e-3;
219 break;
220 case EI:
221 tau = 3e-3;
222 delay = 0.8e-3;
223 break;
224 case EE:
225 tau = 3e-3;
226 delay = 1.5e-3;
227 break;
228 default:
229 assert(false);
230 break;
231 }
232
233 tau_[iEdg] = tau;
234 totalDelay_[iEdg] = static_cast<int>(delay / deltaT) + 1;
235
236 // initializes the queues for the Synapses
237 initSpikeQueue(iEdg);
238 // reset time varying state vars and recompute decay
239 resetEdge(iEdg, deltaT);
240}

248bool AllSpikingSynapses::isSpikeQueue(const BGSIZE iEdg)
249{
250 uint32_t &delayQueue = delayQueue_[iEdg];
251 int &delayIdx = delayIndex_[iEdg];
252 int &ldelayQueue = delayQueueLength_[iEdg];
253
254 bool r = delayQueue & (0x1 << delayIdx);
255 delayQueue &= ~(0x1 << delayIdx);
256 if (++delayIdx >= ldelayQueue) {
257 delayIdx = 0;
258 }
259 return r;
260}

265void AllSpikingSynapses::preSpikeHit(const BGSIZE iEdg)
266{
267 uint32_t &delayQueue = delayQueue_[iEdg];
268 int &delayIdx = delayIndex_[iEdg];
269 int &ldelayQueue = delayQueueLength_[iEdg];
270 int &totalDelay = totalDelay_[iEdg];
271
272 // Add to spike queue
273
274 // calculate index where to insert the spike into delayQueue
275 //LOG4CPLUS_TRACE(edgeLogger_,"delayidx "<<delayIdx<<" totalDelay "<<totalDelay<<" ldelayQueue "<<ldelayQueue);
276 int idx = delayIdx + totalDelay;
277 if (idx >= ldelayQueue) {
278 idx -= ldelayQueue;
279 }
280 if ((delayQueue & (0x1 << idx)) != 0) {
281 LOG4CPLUS_ERROR(edgeLogger_, "Delay Queue Error " << setbase(2) << delayQueue << setbase(10)
282 << " index " << idx << " edge ID " << iEdg);
283 assert(false);
284 }
285
286 // set a spike
287 delayQueue |= (0x1 << idx);
288}

344bool AllSpikingSynapses::updateDecay(const BGSIZE iEdg, const BGFLOAT deltaT)
