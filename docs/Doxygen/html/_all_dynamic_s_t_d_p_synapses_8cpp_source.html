194void AllDynamicSTDPSynapses::changePSR(const BGSIZE iEdg, const BGFLOAT deltaT)
195{
196 BGFLOAT &psr = this->psr_[iEdg];
197 BGFLOAT &W = this->W_[iEdg];
198 BGFLOAT &decay = this->decay_[iEdg];
199 uint64_t &lastSpike = this->lastSpike_[iEdg];
200 BGFLOAT &r = this->r_[iEdg];
201 BGFLOAT &u = this->u_[iEdg];
202 BGFLOAT &D = this->D_[iEdg];
203 BGFLOAT &F = this->F_[iEdg];
204 BGFLOAT &U = this->U_[iEdg];
205
206 // adjust synapse parameters
207 if (lastSpike != ULONG_MAX) {
208 BGFLOAT isi = (g_simulationStep - lastSpike) * deltaT;
209 r = 1 + (r * (1 - u) - 1) * exp(-isi / D);
210 u = U + u * (1 - U) * exp(-isi / F);
211 }
212 psr += ((W / decay) * u * r); // calculate psr
213 lastSpike = g_simulationStep; // record the time of the spike
214}
