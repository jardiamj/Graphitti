105bool ParameterManager::getStringByXpath(string xpath, string &referenceVar)
106{
107 if (!checkDocumentStatus())
108 return false;
109
110 // temp string holds evaluation of value at xpath

115 return false;

117 referenceVar = temp;

128bool ParameterManager::getIntByXpath(string xpath, int &referenceVar)
129{
130 if (!checkDocumentStatus())
131 return false;
132 string tmp;
133 if (!getStringByXpath(xpath, tmp)) {
134 return false;
135 }
136 // Workaround for standard value conversion functions.
137 // stoi() will cast floats to ints.
138 if (regex_match(tmp, regex("\\d+[.]\\d+(e[+-]?\\d+)?f?|\\d+[.]?\\d+(e[+-]?\\d+)?f"))) {
139 cerr << "Parsed parameter is likely a float/double value. "
140 << "Terminating integer cast. Value: " << tmp << endl;
141 return false;
142 } else if (regex_match(tmp, regex(".*[^\\def.]+.*"))) {
143 cerr << "Parsed parameter is likely a string. "
144 << "Terminating integer cast. Value: " << tmp << endl;
145 return false;
146 }
147 try {
148 referenceVar = stoi(tmp);
149 } catch (invalid_argument &arg_exception) {
150 cerr << "Parsed parameter could not be parsed as an integer. Value: " << tmp << endl;
151 return false;
152 } catch (out_of_range &range_exception) {
153 cerr << "Parsed string parameter could not be converted to an integer. Value: " << tmp
154 << endl;
155 return false;
156 }
157 return true;
158}

292bool ParameterManager::getIntVectorByXpath(const string &path, const string &elementName,
293 vector<int> &referenceVar)
294{
295 // Open file using a local XmlDocument object
296 TiXmlDocument xmlDocument(path.c_str());
297 if (!xmlDocument.LoadFile()) {
298 cerr << "Failed to load " << path.c_str() << ":"
299 << "\n\t" << xmlDocument.ErrorDesc() << endl;
300 return false;
301 }
302
303 // Check file for matching element
304 TiXmlNode *xmlNode = nullptr;
305 if ((xmlNode = xmlDocument.FirstChildElement(elementName)) == nullptr) {
306 cerr << "Could not find <" << elementName << "> in vertex list file " << path << endl;
307 return false;
308 }
309
310 // Get list of ints as a string stream
311 std::istringstream valueStream(xmlNode->ToElement()->GetText());
312
313 // Parse integers out of the string and add them to a list
314 int i;
315 while (valueStream.good()) {
316 valueStream >> i;
317 referenceVar.push_back(i);
318 }
319 return true;
320}
